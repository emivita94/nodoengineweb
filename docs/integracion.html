Guía de integración NODO Engine

API REST · Facturación electrónica Paraguay · SET / SIFEN · v1

De cero a primera factura en 5 pasos

1
Obtener las credenciales

En el panel NODO Engine → empresa → Integración ERP encontrás la URL base y la API Key. Guardá ambos en la configuración de tu sistema — no los hardcodees en el código fuente.

' VB.NET — guardar en app.config
' <appSettings>
'   <add key="NodoApiUrl" value="https://sifen-engine-production-f96e.up.railway.app/api/v1"/>
'   <add key="NodoApiKey" value="sk_live_XXXX"/>
' </appSettings>
Dim NODO_URL As String = ConfigurationManager.AppSettings("NodoApiUrl")
Dim NODO_KEY As String = ConfigurationManager.AppSettings("NodoApiKey")
2
Verificar conectividad

Este endpoint no requiere API Key. Si devuelve 200 OK, la API está disponible.

GET/health
// Respuesta: {"ok":true,"version":"0.1.0","ambiente":"prod"}
3
Emitir una factura de prueba (innominado)

El cliente innominado (tipo: 4) no requiere RUC ni CI. Usá una key sk_test_ para no enviar documentos reales a la SET.

Clase NodoApiClient — VB.NET (lista para usar)

' Requiere NuGet: Install-Package Newtonsoft.Json
Imports System.Net.Http
Imports Newtonsoft.Json
Imports Newtonsoft.Json.Linq

Public Class NodoApiClient

  Private Shared ReadOnly _http    As New HttpClient() With {.Timeout = TimeSpan.FromSeconds(60)}
  Private Shared          _baseUrl As String

  Public Shared Sub Inicializar(baseUrl As String, apiKey As String)
    _baseUrl = baseUrl.TrimEnd("/"c)
    _http.DefaultRequestHeaders.Remove("X-API-Key")
    _http.DefaultRequestHeaders.Add("X-API-Key", apiKey)
  End Sub

  Public Shared Async Function EmitirDocumento(payload As Object) As Task(Of ResultadoNodo)
    Dim d = (Await Post("/documentos/erp", payload))("data")
    Return New ResultadoNodo With {
      .CDC = d("cdc").ToString(), .Numero = d("numero").ToString(),
      .Estado = d("estado").ToString(), .SifenCodigo = d("sifen_codigo")?.ToString(),
      .SifenMensaje = d("sifen_mensaje")?.ToString(),
      .Aprobado = d("estado").ToString() = "aprobado"
    }
  End Function

  Public Shared Async Function Consultar(cdc As String) As Task(Of ResultadoNodo)
    Dim d = (Await [Get]("/documentos/" & cdc))("data")
    Return New ResultadoNodo With {
      .CDC = d("cdc").ToString(), .Estado = d("estado").ToString(),
      .SifenMensaje = d("sifen_mensaje")?.ToString(),
      .Aprobado = d("estado").ToString() = "aprobado"
    }
  End Function

  Public Shared Async Function DescargarKude(cdc As String, Optional fmt As String = "a4") As Task(Of Byte())
    Return Await GetBytes($"/documentos/{cdc}/kude?formato={fmt}")
  End Function

  Public Shared Async Function DescargarXML(cdc As String) As Task(Of Byte())
    Return Await GetBytes($"/documentos/{cdc}/xml")
  End Function

  Public Shared Async Function Cancelar(cdc As String, motivo As String) As Task(Of Boolean)
    Return (Await Post($"/documentos/{cdc}/cancelar", New With {.motivo = motivo}))("ok").Value(Of Boolean)()
  End Function

  Private Shared Async Function Post(ep As String, payload As Object) As Task(Of JObject)
    Dim c = New StringContent(JsonConvert.SerializeObject(payload), System.Text.Encoding.UTF8, "application/json")
    Return Await Parse(Await _http.PostAsync(_baseUrl & ep, c))
  End Function

  Private Shared Async Function [Get](ep As String) As Task(Of JObject)
    Return Await Parse(Await _http.GetAsync(_baseUrl & ep))
  End Function

  Private Shared Async Function GetBytes(ep As String) As Task(Of Byte())
    Dim res = Await _http.GetAsync(_baseUrl & ep)
    If Not res.IsSuccessStatusCode Then Throw New Exception("HTTP " & CInt(res.StatusCode))
    Return Await res.Content.ReadAsByteArrayAsync()
  End Function

  Private Shared Async Function Parse(res As HttpResponseMessage) As Task(Of JObject)
    Dim body = Await res.Content.ReadAsStringAsync()
    Dim obj  = JObject.Parse(body)
    If Not res.IsSuccessStatusCode Then
      Throw New Exception("Error API: " & (obj("error")?.ToString() ?? res.StatusCode.ToString()))
    End If
    Return obj
  End Function

End Class

Public Class ResultadoNodo
  Public Property CDC, Numero, Estado, SifenCodigo, SifenMensaje As String
  Public Property Aprobado As Boolean
End Class

' Inicializar UNA SOLA VEZ al arrancar la app:

  ConfigurationManager.AppSettings("NodoApiUrl"),
  ConfigurationManager.AppSettings("NodoApiKey")
)

Emitir una factura electrónica

Genera el XML SIFEN, lo firma con el certificado digital del emisor y lo envía a la SET — en una sola llamada.

POST/api/v1/documentos/erp

Request completo comentado

{
  // TIPO
  "tipoDocumento": 1,              // 1=Factura | 4=Autofactura | 5=NC | 6=ND | 7=Remisión

  // FECHA Y CONDICIONES
  "fecha": "2025-04-03T10:30:00",  // ISO 8601. Si se omite, usa la hora actual del servidor
  "moneda": "PYG",                 // PYG (defecto) | USD | BRL | EUR | ARS
  "condicionVenta": 1,             // 1=Contado (defecto) | 2=Crédito

  // RECEPTOR
  "cliente": {
    "tipo": 1,                      // 1=RUC | 2=CI | 3=Pasaporte/extranjero | 4=Innominado
    "documento": "80012345-6",     // RUC con dígito. NO enviar si tipo=4
    "razonSocial": "Mi Empresa SA", // Nombre completo. NO enviar si tipo=4
    "email": "fe@empresa.py",       // Opcional — NODO envía el KUDE automáticamente al aprobarse
    "telefono": "021555000",        // Opcional
    "direccion": "Av. Principal 123" // Opcional
  },

  // ÍTEMS — mínimo 1
  "items": [
    {
      "descripcion": "Notebook HP 15",
      "cantidad": 2,                 // Puede ser decimal: 1.5
      "precioUnitario": 2500000,     // SIN IVA incluido. Sin decimales en PYG.
      "precioTotal": 5000000,        // cantidad × precioUnitario — el motor valida la coherencia
      "tasaIVA": 10,                 // 0=Exento | 5=IVA 5% | 10=IVA 10%
      "unidadMedida": 77,            // 77=Unidad (defecto) | 83=Servicio | ver tabla
      "codigoProducto": "NB-HP-15"  // Opcional — código interno del ERP
    },
    {
      "descripcion": "Soporte técnico",
      "cantidad": 1, "precioUnitario": 500000, "precioTotal": 500000,
      "tasaIVA": 0   // Exento — en un mismo comprobante puede haber distintas tasas
    }
  ],

  // TRAZABILIDAD
  "referenciaExterna": "FAC-2025-0042", // Tu ID interno — vuelve en el webhook
  "webhookUrl": "https://mi-erp.py/webhook/nodo" // Opcional — NODO notifica cuando la SET responde
}

VB.NET

Public Async Function EmitirFactura(venta As VentaDTO) As Task(Of ResultadoNodo)
  Dim payload = New With {
    .tipoDocumento = 1, .moneda = "PYG",
    .fecha         = venta.Fecha.ToString("yyyy-MM-ddTHH:mm:ss"),
    .condicionVenta = If(venta.EsCredito, 2, 1),
    .cliente = New With {
      .tipo = venta.TipoDocCliente, .documento = venta.RucCiCliente,
      .razonSocial = venta.NombreCliente, .email = venta.EmailCliente
    },
    .items = venta.Detalles.Select(Function(d) New With {
      .descripcion = d.Descripcion, .cantidad = d.Cantidad,
      .precioUnitario = d.PrecioUnitario, .precioTotal = d.PrecioTotal,
      .tasaIVA = d.TasaIVA, .codigoProducto = d.CodigoInterno
    }).ToArray(),
    .referenciaExterna = venta.NroFacturaInterno,
    .webhookUrl = ConfigurationManager.AppSettings("NodoWebhookUrl")
  }
  Try
    Dim r = Await NodoApiClient.EmitirDocumento(payload)
    FacturasDao.ActualizarCDC(venta.NroFacturaInterno, r.CDC, r.Estado)
    If r.Aprobado Then
      Dim tmp = Path.Combine(Path.GetTempPath(), $"KUDE-{r.Numero}.pdf")
      File.WriteAllBytes(tmp, Await NodoApiClient.DescargarKude(r.CDC))
      Process.Start(tmp) ' Abre en Adobe Reader
    Else
      MessageBox.Show("SET rechazó: " & r.SifenMensaje, "Atención", MessageBoxButtons.OK, MessageBoxIcon.Warning)
    End If
    Return r
  Catch ex As Exception
    MessageBox.Show("Error: " & ex.Message) : Return Nothing
  End Try
End Function
Documentos rechazados: cuando la SET rechaza, el HTTP status sigue siendo 201 con estado: "rechazado". Guardar el CDC igualmente. Para corregir y reemitir, crear un documento nuevo — no se puede reenviar el mismo CDC.

Endpoint receptor VB.NET (ASP.NET WebAPI)

<HttpPost> <Route("webhook/nodo")>
Public Async Function RecibirWebhook(payload As JObject) As Task(Of IHttpActionResult)
  ' Responder 200 INMEDIATAMENTE — si tardás más de 10s, NODO lo reintenta
  Task.Run(Sub()
    Try
      Dim doc = payload("documento")
      Dim ref = doc("referenciaExterna")?.ToString()
      Dim cdc = doc("cdc").ToString()
      Select Case payload("evento").ToString()
        Case "de.aprobado"  : FacturasDao.ActualizarEstado(ref, cdc, "APROBADA")
        Case "de.rechazado" : FacturasDao.ActualizarEstado(ref, cdc, "RECHAZADA")
        Case "de.cancelado" : FacturasDao.ActualizarEstado(ref, cdc, "CANCELADA")
      End Select
    Catch ex As Exception : LogError(ex.Message)
    End Try
  End Sub)
  Return Ok()
End Function

Cálculo del IVA

' El motor calcula el IVA automáticamente. NO enviar el monto del IVA.
IVA 10% = precioTotal / 11
IVA  5% = precioTotal / 21

' VB.NET — para mostrar en tu UI:
Function CalcularIVA(total As Decimal, tasa As Integer) As Decimal
  Select Case tasa
    Case 10 : Return Math.Round(total / 11, 0)
    Case  5 : Return Math.Round(total / 21, 0)
    Case  0 : Return 0
    Case Else : Throw New ArgumentException("tasaIVA debe ser 0, 5 ó 10")
  End Select
End Function

Python / PHP / cURL

Python

import requests, os

URL = os.getenv('NODO_API_URL')
HDR = {'X-API-Key': os.getenv('NODO_API_KEY'), 'Content-Type': 'application/json'}

def emitir(cliente, items, ref=None):
    r = requests.post(f'{URL}/documentos/erp',
        json={'tipoDocumento':1,'cliente':cliente,'items':items,'referenciaExterna':ref},
        headers=HDR, timeout=30)
    r.raise_for_status()
    return r.json()['data']

def kude(cdc, ruta='factura.pdf'):

function nodo($ep, $data) {
  $ch = curl_init($_ENV['NODO_URL'].$ep);
  curl_setopt_array($ch,[CURLOPT_POST=>true,CURLOPT_RETURNTRANSFER=>true,CURLOPT_TIMEOUT=>30,
    CURLOPT_POSTFIELDS=>json_encode($data),
    CURLOPT_HTTPHEADER=>['Content-Type: application/json','X-API-Key: '.$_ENV['NODO_KEY']]]);
  $r=json_decode(curl_exec($ch),true); curl_close($ch);
  if(!$r['ok']) throw new Exception($r['error']);
  return $r['data'];
}
